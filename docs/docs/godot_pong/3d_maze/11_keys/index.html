	var used_key = ""
	if to_open:
		var allowed = true
		if key_needed != "":
			allowed = keys.has(key_needed)
		if allowed:
			used_key = key_needed
			key_needed = ""

			door_open = true
	else:
		$AnimationPlayer.play_backwards("open")
		door_open = false
	return used_key
